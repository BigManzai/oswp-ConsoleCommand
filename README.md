Remote Config Opensim.ini:

     [RemoteAdmin]
     enabled = true
     access_password = your secret passwd
     enabled_methods = all

Remote Config Robust.ini:

    ConsoleUser = Name
    ConsolePass = your secret passwd
    ConsolePort = 0
